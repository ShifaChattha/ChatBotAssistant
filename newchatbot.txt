-- Products table
CREATE TABLE IF NOT EXISTS products (
    id INT AUTO_INCREMENT PRIMARY KEY,
    category VARCHAR(50),
    name VARCHAR(100),
    price VARCHAR(10)
);

-- Shades table
CREATE TABLE IF NOT EXISTS shades (
    id INT AUTO_INCREMENT PRIMARY KEY,
    product_id INT,
    shade_name VARCHAR(50),
    stock INT,
    FOREIGN KEY (product_id) REFERENCES products(id)
);
